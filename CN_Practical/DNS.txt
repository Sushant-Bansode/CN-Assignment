import java.net.*;
import java.util.*;

public class dns_new {
    public static void main(String args[]) {
        Scanner sc = new Scanner(System.in);
        int ch;
        System.out.println("\n1.Host to IP Address\n2.IP address to Host\nEnter Your Choice:");
        ch = sc.nextInt();
        if (ch == 1) {
            try {
                System.out.println("\nEnter Host Name :");
                String host = sc.next();
                InetAddress address = InetAddress.getByName(host);
                System.out.println("\nIp Address is :" + address.getHostAddress());
            } catch (Exception ex) {
                System.out.println("\nInvalid Hostname");
            }
        } else if (ch == 2) {
            try {
                System.out.println("\nEnter IP Address :");
                String host = sc.next();
                InetAddress address = InetAddress.getByName(host);
                System.out.println("\nHost Name is :" + address.getHostName());
            } catch (Exception ex) {
                System.out.println("Enter valid IP Address");
            }
        } else {
            System.out.println("Enter Valid Choice");
        }
    }
}
